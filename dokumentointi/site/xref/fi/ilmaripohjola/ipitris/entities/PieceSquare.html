5   /**
6    * a certain kind of subClass of Piece.
7    *
8    * @author omistaja
9    */
10  public class PieceSquare extends Piece {
11  
12      /**
13       * Uses superclasses constructor and in addition set's blocks' x and y to
14       * form a Tetris Square-Piece.
15       *
16       * @param color the color of the piece
17       * @param x the anchor x-value of the Piece
18       * @param y the anchor y-value of the Piece
19       */
20      public PieceSquare(Color color, int x, int y) {
21          super(color, 4, x, y);
22          super.blocks[0] = new Block(color, x, y - 1);
23          super.blocks[1] = new Block(color, x + 1, y - 1);
24          super.blocks[2] = new Block(color, x, y);
25          super.blocks[3] = new Block(color, x + 1, y);
26      }
27  }
