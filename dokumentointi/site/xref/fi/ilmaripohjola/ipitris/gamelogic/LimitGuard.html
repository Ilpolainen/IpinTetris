3   import fi.ilmaripohjola.ipitris.entities.Block;
4   import fi.ilmaripohjola.ipitris.entities.Piece;
5   import fi.ilmaripohjola.ipitris.entities.Table;
6   
7   /**
8    * Check's TetrisLogic's current Piece is in desired limit's.
9    *
10   * @author omistaja
11   */
12  public class LimitGuard {
13  
14      /**
15       * Checks if the given table has indexes given by x- and y- coordinates.
16       *
17       * @param x an int to present the x-value of an object given by caller
18       * @param y an int to present the y-value of an object given by caller
19       * @param table a Table -object that indexes the gives x and y values are
20       * compared to.
21       * @return Returns false if table[y][x] is outOfBounds.
22       */
23      public boolean blockWithinLimits(int x, int y, Table table) {
24          if (x < 0 || x >= table.getWidth()) {
25              return false;
26          }
27          if (y < 0 || y >= table.getHeight()) {
28              return false;
29          }
30          return true;
31      }
32  
33      /**
34       * Calls blockWithinLimits for every block's x and y -coordinate in
35       * current's blockarray.
36       *
37       * @param current a Piece -object whose blocks are compared to given Table's
38       * blocks.
39       * @param table a Table -object whose blocks are compared to given Piece's
40       * blocks-
41       * @return If any of those calls return false. This method returns false.
42       * Otherwise true.
43       */
44      public boolean pieceWithinLimits(Piece current, Table table) {
45          Block[] currentBlocks = current.getBlocks();
46          for (Block block : currentBlocks) {
47              if (!blockWithinLimits(block.getX(), block.getY(), table)) {
48                  return false;
49              }
50          }
51          return true;
52      }
53  
54      /**
55       * Checks if any block in current's -blocarray has the same x- and y- values
56       * with any block in table's -blockarray.
57       *
58       * @param current Piece current given by caller.
59       * @param table Table given by caller.
60       * @return Returns true if some of them have the same values.
61       */
62      public boolean connects(Piece current, Table table) {
63          Block[] currentBlocks = current.getBlocks();
64          for (Block block : currentBlocks) {
65              if (table.getBlocks()[block.getX()][block.getY()] != null) {
66                  return true;
67              }
68          }
69          return false;
70      }
71  }
