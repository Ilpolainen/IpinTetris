7   /**
8    * Helps GameLogic -class to handle row -events like searching and destroying
9    * them.
10   *
11   * @author omistaja
12   */
13  public class RowManager {
14  
15      /**
16       * Calls destroyRow for every row with index in the index-list given by call
17       * searchFullRows, and calls levelmanager to handle scoring and levelling
18       * accordingly.
19       *
20       * @param levelManager The LevelManager whose points are updated while
21       * destroying rows.
22       * @param table The table whose rows are affected.
23       */
24      public void destroyRows(LevelManager levelManager, Table table) {
25          ArrayList<Integer> rowsToDestroy = searchFullRows(levelManager, table);
26          if (rowsToDestroy.size() == 1) {
27              levelManager.increasePoints(1);
28          }
29          if (rowsToDestroy.size() == 2) {
30              levelManager.increasePoints(3);
31          }
32          if (rowsToDestroy.size() == 3) {
33              levelManager.increasePoints(5);
34          }
35          if (rowsToDestroy.size() == 4) {
36              levelManager.increasePoints(8);
37          }
38          for (Integer row : rowsToDestroy) {
39              destroyRow(row, table);
40          }
41      }
42  
43      /**
44       * Set's given row's block array's blocks null. And moves all blocks with
45       * smaller y-coordinate in the table one step down.
46       *
47       * @param i the rownumber to destroy
48       * @param table The Table whose row is to be destroyed.
49       */
50      public void destroyRow(int i, Table table) {
51          Block[][] blocks = table.getBlocks();
52          for (int j = 0; j < table.getWidth(); j++) {
53              blocks[j][i] = null;
54          }
55          for (int j = i; j >= 0; j--) {
56              for (int k = 0; k < table.getWidth(); k++) {
57                  if (blocks[k][j] != null) {
58                      blocks[k][j].moveDown();
59                      blocks[k][j + 1] = table.getBlocks()[k][j];
60                      blocks[k][j] = null;
61                  }
62              }
63          }
64      }
65  
66      /**
67       * Searches every row in Tables block array with no null values, and adds
68       * it's row-index to a list.
69       *
70       * @param levelManager The LevelManager, whose destroyedRows is updated
71       * after collecting the correct rows.
72       * @param table The Table whose rows are effected.
73       * @return The list of indexes of rows with no null block -values.
74       */
75      public ArrayList<Integer> searchFullRows(LevelManager levelManager, Table table) {
76          ArrayList<Integer> rowsToDestroy = new ArrayList<>();
77          int piecesInRow = 0;
78          for (int i = 0; i < table.getHeight(); i++) {
79              for (int j = 0; j < table.getWidth(); j++) {
80                  if (table.getBlocks()[j][i] != null) {
81                      piecesInRow++;
82                  }
83              }
84              if (piecesInRow == table.getWidth()) {
85                  rowsToDestroy.add(i);
86                  levelManager.increaseRowsDestroyed();
87              }
88              piecesInRow = 0;
89          }
90          return rowsToDestroy;
91      }
92  }
