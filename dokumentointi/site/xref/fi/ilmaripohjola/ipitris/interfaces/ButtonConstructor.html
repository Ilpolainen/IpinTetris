8   import javax.swing.JButton;
9   import javax.swing.JPanel;
10  
11  /**
12   *
13   * @author omistaja
14   */
15  public class ButtonConstructor {
16  
17      public ButtonConstructor() {
18      }
19  
20      public void createButtons(StateCoordinator stateCoordinator, JPanel[] panels) {
21          JButton start = new JButton("START");
22          JButton options = new JButton("OPTIONS");
23          JButton trash = new JButton("TRASH");
24          JButton exit = new JButton("EXIT");
25          JButton confirm = new JButton("CONFIRM");
26          JButton boardOptions = new JButton("BOARD OPTIONS");
27          JButton visualOptions = new JButton("VISUAL OPTIONS");
28          JButton back = new JButton("BACK");
29          JButton back2 = new JButton("BACK");
30          JButton colors = new JButton("COLORS");
31          JButton keys = new JButton("KEY OPTIONS");
32          JButton down = new JButton("DOWN");
33          JButton left = new JButton("LEFT");
34          JButton right = new JButton("RIGHT");
35          JButton rotate = new JButton("ROTATE");
36          JButton back3 = new JButton("BACK");
37          attachButtons(panels, start, options, trash, exit, confirm, boardOptions, back, visualOptions, back2, colors, keys, down, left, right, rotate, back3);
38          connectButtons(stateCoordinator, start, options, trash, exit, confirm, boardOptions, back, visualOptions, back2, colors, keys, down, left, right, rotate, back3);
39      }
40  
41      public void attachButtons(JPanel[] panels, JButton start, JButton options, JButton trash, JButton exit, JButton confirm, JButton boardOptions, JButton back, JButton visualOptions, JButton back2, JButton colors, JButton keys, JButton down, JButton left, JButton right, JButton rotate, JButton back3) {
42          panels[1].add(confirm);
43          panels[1].add(boardOptions);
44          panels[1].add(visualOptions);
45          panels[1].add(keys);
46          panels[0].add(start);
47          panels[0].add(options);
48          panels[0].add(trash);
49          panels[0].add(exit);
50          panels[2].add(back);
51          panels[3].add(back2);
52          panels[3].add(colors);
53          JPanel buttons = new JPanel();
54          buttons.add(down);
55          buttons.add(left);
56          buttons.add(right);
57          buttons.add(rotate);
58          panels[5].add(back3);
59          panels[5].add(buttons);
60      }
61  
62      public void connectButtons(StateCoordinator stateCoordinator, JButton start, JButton options, JButton trash, JButton exit, JButton confirm, JButton boardOptions, JButton back, JButton visualOptions, JButton back2, JButton colors, JButton buttons, JButton down, JButton left, JButton right, JButton rotate, JButton back3) {
63          start.addActionListener(stateCoordinator);
64          exit.addActionListener(stateCoordinator);
65          trash.addActionListener(stateCoordinator);
66          options.addActionListener(stateCoordinator);
67          confirm.addActionListener(stateCoordinator);
68          boardOptions.addActionListener(stateCoordinator);
69          back.addActionListener(stateCoordinator);
70          visualOptions.addActionListener(stateCoordinator);
71          back2.addActionListener(stateCoordinator);
72          colors.addActionListener(stateCoordinator);
73          buttons.addActionListener(stateCoordinator);
74          down.addActionListener(stateCoordinator);
75          left.addActionListener(stateCoordinator);
76          right.addActionListener(stateCoordinator);
77          rotate.addActionListener(stateCoordinator);
78          back3.addActionListener(stateCoordinator);
79      }
