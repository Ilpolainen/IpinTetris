22      }

24      public void createSliders(StartingScreen ss) {
25          JLabel widthName = new JLabel("    WIDTH:");
26          JLabel heightName = new JLabel("    HEIGHT:");
27          JLabel scaleName = new JLabel("     SCALE:");
28          setFonts(widthName, heightName, scaleName);
29          JSlider widthSlider = new JSlider(JSlider.HORIZONTAL, 3, 40, ss.getWidth());
30          JSlider heightSlider = new JSlider(JSlider.HORIZONTAL, 4, 40, ss.getHeight());
31          JSlider scaleSlider = new JSlider(JSlider.HORIZONTAL, 2, 80, ss.getScale());
32          widthSlider.setFocusable(false);
33          heightSlider.setFocusable(false);
34          scaleSlider.setFocusable(false);
35          widthSlider.setName("width");
36          heightSlider.setName("height");
37          scaleSlider.setName("scale");
38          setUpSliders(ss, widthSlider, heightSlider, scaleSlider);
39          JPanel[] panels = ss.getPanels();
40          panels[2].add(widthName);
41          panels[2].add(widthSlider);
42          panels[2].add(heightName);
43          panels[2].add(heightSlider);
44          panels[3].add(scaleName);
45          panels[3].add(scaleSlider);
46      }
47  
48      /**
49       * Sets up the options sliders for the startingscreens JFrame. Defins Minor-
50       * and Major- TickSpacing, sets labels and ticks visible and adds
51       * actionlistener for them.
52       *
53       * @param ss StartingScreen
54       * @param width JSlider widthSlider
55       * @param height JSlider height Slider
56       * @param scale JSlider Scale Slider
57       */
58      private void setUpSliders(StartingScreen ss, JSlider width, JSlider height, JSlider scale) {
59          width.setMinorTickSpacing(1);
60          width.setMajorTickSpacing(1);
61          height.setMinorTickSpacing(1);
62          height.setMajorTickSpacing(1);
63          scale.setMinorTickSpacing(1);
64          scale.setMajorTickSpacing(1);
65          width.setPaintTicks(true);
66          height.setPaintTicks(true);
67          scale.setPaintTicks(true);
68          width.setPaintLabels(true);
69          height.setPaintLabels(true);
70          scale.setPaintLabels(true);
71          this.createLabelTable(width);
72          this.createLabelTable(height);
73          this.createLabelTable(scale);
74          width.addChangeListener((ChangeListener) ss.getStateCoordinator().getSliderCoordinator());
75          height.addChangeListener((ChangeListener) ss.getStateCoordinator().getSliderCoordinator());
76          scale.addChangeListener((ChangeListener) ss.getStateCoordinator().getSliderCoordinator());
77      }
78  
79      /**
80       * Creates labels for the slider given as a parameter.
81       *
82       * @param slider slider on which the labeltable is assigned
83       */
84      private void createLabelTable(JSlider slider) {
85          Hashtable lableTable = new Hashtable();
86          lableTable.put(5, new JLabel("5"));
87          lableTable.put(10, new JLabel("10"));
88          lableTable.put(15, new JLabel("15"));
89          lableTable.put(20, new JLabel("20"));
90          lableTable.put(25, new JLabel("25"));
91          lableTable.put(30, new JLabel("30"));
92          lableTable.put(35, new JLabel("35"));
93          lableTable.put(40, new JLabel("40"));
94          lableTable.put(45, new JLabel("45"));
95          lableTable.put(50, new JLabel("50"));
96          lableTable.put(55, new JLabel("55"));
97          lableTable.put(60, new JLabel("60"));
98          lableTable.put(65, new JLabel("65"));
99          lableTable.put(70, new JLabel("70"));
100         lableTable.put(75, new JLabel("75"));
101         lableTable.put(80, new JLabel("80"));
102         slider.setLabelTable(lableTable);
103     }
104 
105     private void setFonts(JLabel widthName, JLabel heightName, JLabel scaleName) {
106         Font font = new Font("Greek", 1, 12);
107         widthName.setFont(font);
108         heightName.setFont(font);
109         scaleName.setFont(font);
110     }
